```go
package session

import (
	"fmt"
	"strconv"
	"strings"
)

const (
	xorKey   = 0x5A3C7E1D
	codeLen  = 6
	maxBits  = 31
)

// Compile-time assertion: xorKey must fit in maxBits
var _ = uint(xorKey >> maxBits) // panics at compile time if xorKey has bit 31+ set

// bitPerm is a fixed bit permutation table for obfuscation.
// Shuffles bit positions 0-30 to spread sequential changes.
var bitPerm = [31]int{
	17, 5, 23, 11, 29, 2, 19, 8, 26, 14, 1,
	20, 7, 25, 13, 30, 3, 21, 9, 27, 15, 0,
	18, 6, 24, 12, 28, 4, 22, 10, 16,
}

var bitPermInv [31]int

func init() {
	for i, p := range bitPerm {
		bitPermInv[p] = i
	}
}

func shuffleBits(n int, perm [31]int) int {
	var result int
	for i := 0; i < maxBits; i++ {
		if n&(1<<i) != 0 {
			result |= 1 << perm[i]
		}
	}
	return result
}

// EncodeSessionID converts a tmux session ID ("$N") to a 6-char base36 code.
func EncodeSessionID(tmuxID string) (string, error) {
	if !strings.HasPrefix(tmuxID, "$") || len(tmuxID) < 2 {
		return "", fmt.Errorf("invalid tmux session ID: %q", tmuxID)
	}
	n, err := strconv.Atoi(tmuxID[1:])
	if err != nil {
		return "", fmt.Errorf("invalid tmux session ID: %q: %w", tmuxID, err)
	}
	if n < 0 || n >= (1<<maxBits) {
		return "", fmt.Errorf("tmux session ID out of range: %d", n)
	}
	scrambled := shuffleBits(n^xorKey, bitPerm)
	code := strconv.FormatInt(int64(scrambled), 36)
	// Pad to 6 chars
	for len(code) < codeLen {
		code = "0" + code
	}
	return code, nil
}

// DecodeSessionID converts a 6-char base36 code back to a tmux session ID ("$N").
func DecodeSessionID(code string) (string, error) {
	if len(code) != codeLen {
		return "", fmt.Errorf("invalid session code: %q (must be %d chars)", code, codeLen)
	}
	scrambled, err := strconv.ParseInt(code, 36, 64)
	if err != nil {
		return "", fmt.Errorf("invalid session code: %q: %w", code, err)
	}
	n := shuffleBits(int(scrambled), bitPermInv) ^ xorKey
	if n < 0 {
		return "", fmt.Errorf("decoded negative value from code: %q", code)
	}
	return fmt.Sprintf("$%d", n), nil
}
```

```go
package core

import "fmt"

// ServiceRegistry allows modules to register and discover capabilities by name.
type ServiceRegistry struct {
	services map[string]any
}

func NewServiceRegistry() *ServiceRegistry {
	return &ServiceRegistry{services: make(map[string]any)}
}

func (r *ServiceRegistry) Register(name string, svc any) {
	r.services[name] = svc
}

func (r *ServiceRegistry) Get(name string) (any, bool) {
	svc, ok := r.services[name]
	return svc, ok
}

func (r *ServiceRegistry) MustGet(name string) any {
	svc, ok := r.services[name]
	if !ok {
		panic(fmt.Sprintf("service %q not registered", name))
	}
	return svc
}
```

```go
package core

import (
	"context"
	"fmt"
	"net/http"

	"github.com/wake/tmux-box/internal/config"
	"github.com/wake/tmux-box/internal/tmux"
)

// Module is the interface all daemon modules implement.
type Module interface {
	Name() string
	Init(core *Core) error
	RegisterRoutes(mux *http.ServeMux)
	Start(ctx context.Context) error
	Stop() error
}

// CoreDeps holds the shared infrastructure injected into Core.
type CoreDeps struct {
	Config   *config.Config
	Tmux     tmux.Executor
	Registry *ServiceRegistry
}

// Core holds shared infrastructure and manages module lifecycle.
type Core struct {
	Cfg      *config.Config
	Tmux     tmux.Executor
	Registry *ServiceRegistry
	modules  []Module
}

func New(deps CoreDeps) *Core {
	reg := deps.Registry
	if reg == nil {
		reg = NewServiceRegistry()
	}
	return &Core{
		Cfg:      deps.Config,
		Tmux:     deps.Tmux,
		Registry: reg,
	}
}

func (c *Core) AddModule(m Module) {
	c.modules = append(c.modules, m)
}

func (c *Core) InitModules() error {
	for _, m := range c.modules {
		if err := m.Init(c); err != nil {
			return fmt.Errorf("module %s init: %w", m.Name(), err)
		}
	}
	return nil
}

func (c *Core) RegisterRoutes(mux *http.ServeMux) {
	for _, m := range c.modules {
		m.RegisterRoutes(mux)
	}
}

func (c *Core) StartModules(ctx context.Context) error {
	for _, m := range c.modules {
		if err := m.Start(ctx); err != nil {
			return fmt.Errorf("module %s start: %w", m.Name(), err)
		}
	}
	return nil
}

func (c *Core) StopModules() error {
	// Stop in reverse order
	for i := len(c.modules) - 1; i >= 0; i-- {
		if err := c.modules[i].Stop(); err != nil {
			return fmt.Errorf("module %s stop: %w", c.modules[i].Name(), err)
		}
	}
	return nil
}
```

```go
package store

import (
	"database/sql"
	_ "modernc.org/sqlite"
)

// SessionMeta is the DB representation of session meta cache.
type SessionMeta struct {
	TmuxID      string
	Mode        string
	CCSessionID string
	CCModel     string
	Cwd         string
}

// MetaUpdate supports partial updates (nil = no change).
type MetaUpdate struct {
	Mode        *string
	CCSessionID *string
	CCModel     *string
	Cwd         *string
}

// MetaStore manages the session_meta table.
type MetaStore struct {
	db *sql.DB
}

func OpenMeta(path string) (*MetaStore, error) {
	db, err := sql.Open("sqlite", path+"?_pragma=journal_mode(wal)")
	if err != nil {
		return nil, err
	}
	ms := &MetaStore{db: db}
	if err := ms.migrate(); err != nil {
		db.Close()
		return nil, err
	}
	return ms, nil
}

func (ms *MetaStore) Close() error {
	return ms.db.Close()
}

func (ms *MetaStore) migrate() error {
	_, err := ms.db.Exec(`
		CREATE TABLE IF NOT EXISTS session_meta (
			tmux_id       TEXT PRIMARY KEY,
			mode          TEXT DEFAULT 'term',
			cc_session_id TEXT DEFAULT '',
			cc_model      TEXT DEFAULT '',
			cwd           TEXT DEFAULT '',
			created_at    DATETIME DEFAULT CURRENT_TIMESTAMP
		)
	`)
	return err
}

func (ms *MetaStore) SetMeta(tmuxID string, meta SessionMeta) error {
	_, err := ms.db.Exec(`
		INSERT INTO session_meta (tmux_id, mode, cc_session_id, cc_model, cwd)
		VALUES (?, ?, ?, ?, ?)
		ON CONFLICT(tmux_id) DO UPDATE SET
			mode=excluded.mode,
			cc_session_id=excluded.cc_session_id,
			cc_model=excluded.cc_model,
			cwd=excluded.cwd
	`, tmuxID, meta.Mode, meta.CCSessionID, meta.CCModel, meta.Cwd)
	return err
}

func (ms *MetaStore) GetMeta(tmuxID string) (*SessionMeta, error) {
	row := ms.db.QueryRow(`
		SELECT tmux_id, mode, cc_session_id, cc_model, cwd
		FROM session_meta WHERE tmux_id = ?
	`, tmuxID)
	var m SessionMeta
	err := row.Scan(&m.TmuxID, &m.Mode, &m.CCSessionID, &m.CCModel, &m.Cwd)
	if err == sql.ErrNoRows {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	return &m, nil
}

func (ms *MetaStore) ListMeta() ([]SessionMeta, error) {
	rows, err := ms.db.Query(`SELECT tmux_id, mode, cc_session_id, cc_model, cwd FROM session_meta`)
	if err != nil {
		return nil, err
	}
	defer rows.Close()
	var metas []SessionMeta
	for rows.Next() {
		var m SessionMeta
		if err := rows.Scan(&m.TmuxID, &m.Mode, &m.CCSessionID, &m.CCModel, &m.Cwd); err != nil {
			return nil, err
		}
		metas = append(metas, m)
	}
	return metas, nil
}

func (ms *MetaStore) UpdateMeta(tmuxID string, update MetaUpdate) error {
	// Ensure row exists first
	existing, err := ms.GetMeta(tmuxID)
	if err != nil {
		return err
	}
	if existing == nil {
		// Create with defaults, then apply update
		existing = &SessionMeta{TmuxID: tmuxID, Mode: "term"}
	}
	if update.Mode != nil {
		existing.Mode = *update.Mode
	}
	if update.CCSessionID != nil {
		existing.CCSessionID = *update.CCSessionID
	}
	if update.CCModel != nil {
		existing.CCModel = *update.CCModel
	}
	if update.Cwd != nil {
		existing.Cwd = *update.Cwd
	}
	return ms.SetMeta(tmuxID, *existing)
}

func (ms *MetaStore) DeleteMeta(tmuxID string) error {
	_, err := ms.db.Exec(`DELETE FROM session_meta WHERE tmux_id = ?`, tmuxID)
	return err
}

// CleanOrphans deletes meta for tmux IDs not in the given live set.
// Returns the number of removed entries.
func (ms *MetaStore) CleanOrphans(liveTmuxIDs []string) (int, error) {
	metas, err := ms.ListMeta()
	if err != nil {
		return 0, err
	}
	live := make(map[string]bool, len(liveTmuxIDs))
	for _, id := range liveTmuxIDs {
		live[id] = true
	}
	removed := 0
	for _, m := range metas {
		if !live[m.TmuxID] {
			if err := ms.DeleteMeta(m.TmuxID); err != nil {
				return removed, err
			}
			removed++
		}
	}
	return removed, nil
}

// ResetStaleModes resets all non-term sessions to term on startup.
func (ms *MetaStore) ResetStaleModes() error {
	_, err := ms.db.Exec(`UPDATE session_meta SET mode = 'term' WHERE mode != 'term'`)
	return err
}
```

```go
type createRequest struct {
	Name string `json:"name"`
	Cwd  string `json:"cwd,omitempty"`
}

func (m *SessionModule) handleCreate(w http.ResponseWriter, r *http.Request) {
	var req createRequest
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "invalid JSON", 400)
		return
	}
	if !validSessionName.MatchString(req.Name) {
		http.Error(w, "invalid session name", 400)
		return
	}
	cwd := req.Cwd
	if cwd == "" {
		cwd = "~"
	}
	if err := m.tmux.NewSession(req.Name, cwd); err != nil {
		http.Error(w, "failed to create tmux session: "+err.Error(), 500)
		return
	}
	// Find the newly created session to get its tmux ID
	sessions, err := m.tmux.ListSessions()
	if err != nil {
		http.Error(w, err.Error(), 500)
		return
	}
	for _, ts := range sessions {
		if ts.Name == req.Name {
			code, _ := EncodeSessionID(ts.ID)
			m.meta.SetMeta(ts.ID, store.SessionMeta{Mode: "term", Cwd: cwd})
			info := SessionInfo{
				Code: code, TmuxID: ts.ID, Name: ts.Name,
				Exists: true, Mode: "term", Cwd: cwd,
			}
			w.Header().Set("Content-Type", "application/json")
			w.WriteHeader(201)
			json.NewEncoder(w).Encode(info)
			return
		}
	}
	http.Error(w, "session created but not found", 500)
}

		http.Error(w, err.Error(), 500)
		return

		http.Error(w, "session not found", 404)

	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "invalid JSON", 400)
		return
	}

	case "term", "stream", "jsonl":
	default:

		return
	}
	if err := m.UpdateMeta(code, MetaUpdate{Mode: &req.Mode}); err != nil {
		http.Error(w, err.Error(), 500)
		return
	}
	w.WriteHeader(200)
	json.NewEncoder(w).Encode(map[string]string{"mode": req.Mode})
}
```

```go
func (ms *MetaStore) MigrateFromLegacy(legacyDB *sql.DB, tmuxSessions []tmux.TmuxSession) error {
	if legacyDB == nil || len(tmuxSessions) == 0 {
		return nil // no legacy data or tmux unavailable
	}

	// Build name→tmuxID map
	nameToID := make(map[string]string)
	for _, ts := range tmuxSessions {
		nameToID[ts.Name] = ts.ID
	}

	rows, err := legacyDB.Query(`SELECT name, mode, cc_session_id, cc_model, cwd FROM sessions`)
	if err != nil {
		return nil // legacy table might not exist
	}
	defer rows.Close()

	for rows.Next() {
		var name, mode, ccID, ccModel, cwd string
		rows.Scan(&name, &mode, &ccID, &ccModel, &cwd)
		tmuxID, ok := nameToID[name]
		if !ok {
			continue // session no longer exists
		}
		ms.SetMeta(tmuxID, SessionMeta{
			Mode: mode, CCSessionID: ccID, CCModel: ccModel, Cwd: cwd,
		})
	}
	return nil
}
```
